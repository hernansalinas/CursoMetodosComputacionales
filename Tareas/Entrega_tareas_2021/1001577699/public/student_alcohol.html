In just one apply, we need to extract the Dois of Davide Gaiotto’s articles: 

def get_author(row):
    try:
        if any([ row.get('author')[i]['given'] == 'Tatsuo' and
                  row.get('author')[i]['family'] == 'Kobayashi'
                  for i in range(len(row.get('author'))) ]):
             return row.get('DOI')
        else:
             return None
    except KeyError:
        return None
